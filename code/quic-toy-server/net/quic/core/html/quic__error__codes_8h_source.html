15 enum QuicRstStreamErrorCode {
16  // Complete response has been sent, sending a RST to ask the other endpoint
17  // to stop sending request data without discarding the response.
18  QUIC_STREAM_NO_ERROR = 0,
19 
20  // There was some error which halted stream processing.
21  QUIC_ERROR_PROCESSING_STREAM,
22  // We got two fin or reset offsets which did not match.
23  QUIC_MULTIPLE_TERMINATION_OFFSETS,
24  // We got bad payload and can not respond to it at the protocol level.
25  QUIC_BAD_APPLICATION_PAYLOAD,
26  // Stream closed due to connection error. No reset frame is sent when this
27  // happens.
28  QUIC_STREAM_CONNECTION_ERROR,
29  // GoAway frame sent. No more stream can be created.
30  QUIC_STREAM_PEER_GOING_AWAY,
31  // The stream has been cancelled.
32  QUIC_STREAM_CANCELLED,
33  // Closing stream locally, sending a RST to allow for proper flow control
34  // accounting. Sent in response to a RST from the peer.
35  QUIC_RST_ACKNOWLEDGEMENT,
36  // Receiver refused to create the stream (because its limit on open streams
37  // has been reached). The sender should retry the request later (using
38  // another stream).
39  QUIC_REFUSED_STREAM,
40  // Invalid URL in PUSH_PROMISE request header.
41  QUIC_INVALID_PROMISE_URL,
42  // Server is not authoritative for this URL.
43  QUIC_UNAUTHORIZED_PROMISE_URL,
44  // Can't have more than one active PUSH_PROMISE per URL.
45  QUIC_DUPLICATE_PROMISE_URL,
46  // Vary check failed.
47  QUIC_PROMISE_VARY_MISMATCH,
48  // Only GET and HEAD methods allowed.
49  QUIC_INVALID_PROMISE_METHOD,
50  // The push stream is unclaimed and timed out.
51  QUIC_PUSH_STREAM_TIMED_OUT,
52  // Received headers were too large.
53  QUIC_HEADERS_TOO_LARGE,
54  // The data is not likely arrive in time.
55  QUIC_STREAM_TTL_EXPIRED,
56  // No error. Used as bound while iterating.
57  QUIC_STREAM_LAST_ERROR,
58 };
