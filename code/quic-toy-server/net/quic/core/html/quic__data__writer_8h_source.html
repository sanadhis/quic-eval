20 // This class provides facilities for packing QUIC data.
21 //
22 // The QuicDataWriter supports appending primitive values (int, string, etc)
23 // to a frame instance. The internal memory buffer is exposed as the "data"
24 // of the QuicDataWriter.
25 class QUIC_EXPORT_PRIVATE QuicDataWriter {
26  public:
27  // Creates a QuicDataWriter where |buffer| is not owned.
28  QuicDataWriter(size_t size,
29  char* buffer,
30  Endianness endianness);
31 
32  ~QuicDataWriter();
33 
34  // Returns the size of the QuicDataWriter's data.
35  size_t length() const { return length_; }
36 
37  // Retrieves the buffer from the QuicDataWriter without changing ownership.
38  char* data();
39 
40  // Methods for adding to the payload. These values are appended to the end
41  // of the QuicDataWriter payload.
42 
43  // Writes 8/16/32/64-bit unsigned integers.
44  bool WriteUInt8(uint8_t value);
45  bool WriteUInt16(uint16_t value);
46  bool WriteUInt32(uint32_t value);
47  bool WriteUInt64(uint64_t value);
48 
49  // Write an unsigned-integer value per the IETF QUIC/Variable Length
50  // Integer encoding rules (see draft-ietf-quic-transport-08.txt).
51  // IETF Variable Length Integers have 62 significant bits, so the
52  // value to write must be in the range of 0...(2^62)-1. Returns
53  // false if the value is out of range or if there is no room in the
54  // buffer.
55  bool WriteVarInt62(uint64_t value);
56 
57  // Writes |value| to the position |offset| from the start of the data.
58  // |offset| must be less than the current length of the writer.
59  bool WriteUInt8AtOffset(uint8_t value, size_t offset);
60 
61  // Writes least significant |num_bytes| of a 64-bit unsigned integer in the
62  // correct byte order.
63  bool WriteBytesToUInt64(size_t num_bytes, uint64_t value);
64 
65  // Write unsigned floating point corresponding to the value. Large values are
66  // clamped to the maximum representable (kUFloat16MaxValue). Values that can
67  // not be represented directly are rounded down.
68  bool WriteUFloat16(uint64_t value);
69  bool WriteStringPiece16(QuicStringPiece val);
70  bool WriteBytes(const void* data, size_t data_len);
71  bool WriteRepeatedByte(uint8_t byte, size_t count);
72  // Fills the remaining buffer with null characters.
73  void WritePadding();
74  // Write padding of |count| bytes.
75  bool WritePaddingBytes(size_t count);
76 
77  // Write connection ID as a 64-bit unsigned integer to the payload.
78  // TODO(fayang): Remove this method and use WriteUInt64() once deprecating
79  // quic_restart_flag_quic_rw_cid_in_big_endian and QuicDataWriter has a mode
80  // indicating writing in little/big endian.
81  bool WriteConnectionId(uint64_t connection_id);
82 
83  // Write tag as a 32-bit unsigned integer to the payload. As tags are already
84  // converted to big endian (e.g., CHLO is 'C','H','L','O') in memory by TAG or
85  // MakeQuicTag and tags are written in byte order, so tags on the wire are
86  // in big endian.
87  bool WriteTag(uint32_t tag);
88 
89  size_t capacity() const { return capacity_; }
90 
91  size_t remaining() const { return capacity_ - length_; }
92 
93  private:
94  // Returns the location that the data should be written at, or nullptr if
95  // there is not enough room. Call EndWrite with the returned offset and the
96  // given length to pad out for the next write.
97  char* BeginWrite(size_t length);
98 
99  // TODO(fkastenholz, b/73004262) change buffer_, et al, to be uint8_t, not
100  // char.
101  char* buffer_;
102  size_t capacity_; // Allocation size of payload (or -1 if buffer is const).
103  size_t length_; // Current length of the buffer.
104 
105  // The endianness to write integers and floating numbers.
106  Endianness endianness_;
107 
108  DISALLOW_COPY_AND_ASSIGN(QuicDataWriter);
109 };
