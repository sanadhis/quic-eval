42  // The client uses these objects to keep track of any data to resend upon
43  // receipt of a stateless reject. Recall that the client API allows callers
44  // to optimistically send data to the server prior to handshake-confirmation.
45  // If the client subsequently receives a stateless reject, it must tear down
46  // its existing session, create a new session, and resend all previously sent
47  // data. It uses these objects to keep track of all the sent data, and to
48  // resend the data upon a subsequent connection.
49  class QuicDataToResend {
50  public:
51  // |headers| may be null, since it's possible to send data without headers.
52  QuicDataToResend(std::unique_ptr<SpdyHeaderBlock> headers,
53  QuicStringPiece body,
54  bool fin);

60  virtual void Resend() = 0;
61 
62  protected:
63  std::unique_ptr<SpdyHeaderBlock> headers_;
64  QuicStringPiece body_;
65  bool fin_;
66 
67  private:
68  DISALLOW_COPY_AND_ASSIGN(QuicDataToResend);
69  };
