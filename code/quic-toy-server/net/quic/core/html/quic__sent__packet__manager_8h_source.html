43 // Class which tracks the set of packets sent on a QUIC connection and contains
44 // a send algorithm to decide when to send new packets. It keeps track of any
45 // retransmittable data associated with each packet. If a packet is
46 // retransmitted, it will keep track of each version of a packet so that if a
47 // previous transmission is acked, the data will not be retransmitted.

264  // The retransmission timer is a single timer which switches modes depending
265  // upon connection state.
266  enum RetransmissionTimeoutMode {
267  // A conventional TCP style RTO.
268  RTO_MODE,
269  // A tail loss probe. By default, QUIC sends up to two before RTOing.
270  TLP_MODE,
271  // Retransmission of handshake packets prior to handshake completion.
272  HANDSHAKE_MODE,
273  // Re-invoke the loss detection when a packet is not acked before the
274  // loss detection algorithm expects.
275  LOSS_MODE,
276  };
