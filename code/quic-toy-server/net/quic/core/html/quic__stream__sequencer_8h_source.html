26 // Buffers frames until we have something which can be passed
27 // up to the next layer.
28 class QUIC_EXPORT_PRIVATE QuicStreamSequencer {
29  public:
30  QuicStreamSequencer(QuicStream* quic_stream, const QuicClock* clock);
31  virtual ~QuicStreamSequencer();
32 
33  // If the frame is the next one we need in order to process in-order data,
34  // ProcessData will be immediately called on the stream until all buffered
35  // data is processed or the stream fails to consume data. Any unconsumed
36  // data will be buffered. If the frame is not the next in line, it will be
37  // buffered.
38  void OnStreamFrame(const QuicStreamFrame& frame);
39 
40  // Once data is buffered, it's up to the stream to read it when the stream
41  // can handle more data. The following three functions make that possible.
42 
43  // Fills in up to iov_len iovecs with the next readable regions. Returns the
44  // number of iovs used. Non-destructive of the underlying data.
45  int GetReadableRegions(iovec* iov, size_t iov_len) const;
46 
47  // Fills in one iovec with the next readable region. |timestamp| is
48  // data arrived at the sequencer, and is used for measuring head of
49  // line blocking (HOL). Returns false if there is no readable
50  // region available.
51  bool GetReadableRegion(iovec* iov, QuicTime* timestamp) const;
52 
53  // Copies the data into the iov_len buffers provided. Returns the number of
54  // bytes read. Any buffered data no longer in use will be released.
55  // TODO(rch): remove this method and instead implement it as a helper method
56  // based on GetReadableRegions and MarkConsumed.
57  int Readv(const struct iovec* iov, size_t iov_len);
58 
59  // Consumes |num_bytes| data. Used in conjunction with |GetReadableRegions|
60  // to do zero-copy reads.
61  void MarkConsumed(size_t num_bytes);
62 
63  // Returns true if the sequncer has bytes available for reading.
64  bool HasBytesToRead() const;
65 
66  // Number of bytes available to read.
67  size_t ReadableBytes() const;
68 
69  // Returns true if the sequencer has delivered the fin.
70  bool IsClosed() const;
71 
72  // Calls |OnDataAvailable| on |stream_| if there is buffered data that can
73  // be processed, and causes |OnDataAvailable| to be called as new data
74  // arrives.
75  void SetUnblocked();
76 
77  // Blocks processing of frames until |SetUnblocked| is called.
78  void SetBlockedUntilFlush();
79 
80  // Sets the sequencer to discard all incoming data itself and not call
81  // |stream_->OnDataAvailable()|. |stream_->OnFinRead()| will be called
82  // automatically when the FIN is consumed (which may be immediately).
83  void StopReading();
84 
85  // Free the memory of underlying buffer.
86  void ReleaseBuffer();
87 
88  // Free the memory of underlying buffer when no bytes remain in it.
89  void ReleaseBufferIfEmpty();
90 
91  // Number of bytes in the buffer right now.
92  size_t NumBytesBuffered() const;
93 
94  // Number of bytes has been consumed.
95  QuicStreamOffset NumBytesConsumed() const;
96 
97  int num_frames_received() const { return num_frames_received_; }
98 
99  int num_duplicate_frames_received() const {
100  return num_duplicate_frames_received_;
101  }
102 
103  bool ignore_read_data() const { return ignore_read_data_; }
104 
105  void set_level_triggered(bool level_triggered) {
106  level_triggered_ = level_triggered;
107  }
108 
109  bool level_triggered() const { return level_triggered_; }
110 
111  // Returns std::string describing internal state.
112  const QuicString DebugString() const;
113 
114  private:
115  friend class test::QuicStreamSequencerPeer;
116 
117  // Deletes and records as consumed any buffered data that is now in-sequence.
118  // (To be called only after StopReading has been called.)
119  void FlushBufferedFrames();
120 
121  // Wait until we've seen 'offset' bytes, and then terminate the stream.
122  void CloseStreamAtOffset(QuicStreamOffset offset);
123 
124  // If we've received a FIN and have processed all remaining data, then inform
125  // the stream of FIN, and clear buffers.
126  bool MaybeCloseStream();
127 
128  // The stream which owns this sequencer.
129  QuicStream* stream_;
130 
131  // Stores received data in offset order.
132  QuicStreamSequencerBuffer buffered_frames_;
133 
134  // The offset, if any, we got a stream termination for. When this many bytes
135  // have been processed, the sequencer will be closed.
136  QuicStreamOffset close_offset_;
137 
138  // If true, the sequencer is blocked from passing data to the stream and will
139  // buffer all new incoming data until FlushBufferedFrames is called.
140  bool blocked_;
141 
142  // Count of the number of frames received.
143  int num_frames_received_;
144 
145  // Count of the number of duplicate frames received.
146  int num_duplicate_frames_received_;
147 
148  // Not owned.
149  const QuicClock* clock_;
150 
151  // If true, all incoming data will be discarded.
152  bool ignore_read_data_;
153 
154  // If false, only call OnDataAvailable() when it becomes newly unblocked.
155  // Otherwise, call OnDataAvailable() when number of readable bytes changes.
156  bool level_triggered_;
157

159 };
