70 class QUIC_EXPORT_PRIVATE QuicCryptoServerStream
71  : public QuicCryptoServerStreamBase {
72  public:
73  // QuicCryptoServerStream creates a HandshakerDelegate at construction time
74  // based on the QuicTransportVersion of the connection. Different
75  // HandshakerDelegates provide implementations of different crypto handshake
76  // protocols. Currently QUIC crypto is the only protocol implemented; a future
77  // HandshakerDelegate will use TLS as the handshake protocol.
78  // QuicCryptoServerStream delegates all of its public methods to its
79  // HandshakerDelegate.
80  //
81  // This setup of the crypto stream delegating its implementation to the
82  // handshaker results in the handshaker reading and writing bytes on the
83  // crypto stream, instead of the handshake rpassing the stream bytes to send.
84  class QUIC_EXPORT_PRIVATE HandshakerDelegate {
85  public:
86  virtual ~HandshakerDelegate() {}
87

89  // client hello.

93  // SHA-256 hash of the client's ChannelID key and returns true, if the
94  // client presented a ChannelID. Otherwise it returns false.

111  // NOTE: Indicating that the Expect-CT header should be sent here presents a
