16 // A QuicTag is a 32-bit used as identifiers in the QUIC handshake. The use of
17 // a uint32_t seeks to provide a balance between the tyranny of magic number
18 // registries and the verbosity of std::strings. As far as the wire protocol is
19 // concerned, these are opaque, 32-bit values.
20 //
21 // Tags will often be referred to by their ASCII equivalent, e.g. EXMP. This is
22 // just a mnemonic for the value 0x504d5845 (little-endian version of the ASCII
23 // std::string E X M P).
24 typedef uint32_t QuicTag;
25 typedef std::map<QuicTag, QuicString> QuicTagValueMap;
26 typedef std::vector<QuicTag> QuicTagVector;

36 // Sets |out_result| to the first tag in |our_tags| that is also in |their_tags|
37 // and returns true. If there is no intersection it returns false.

39 // If |out_index| is non-nullptr and a match is found then the index of that
