19 // Class which tracks unacked packets for three purposes:
20 // 1) Track retransmittable data, including multiple transmissions of frames.
21 // 2) Track packets and bytes in flight for congestion control.
22 // 3) Track sent time of packets to provide RTT measurements from acks.
23 class QUIC_EXPORT_PRIVATE QuicUnackedPacketMap {
24  public:
25  QuicUnackedPacketMap();
26  ~QuicUnackedPacketMap();

28  // Adds |serialized_packet| to the map and marks it as sent at |sent_time|.
29  // Marks the packet as in flight if |set_in_flight| is true.
30  // Packets marked as in flight are expected to be marked as missing when they
31  // don't arrive, indicating the need for retransmission.
32  // |old_packet_number| is the packet number of the previous transmission,
33  // or 0 if there was none.
34  // Any AckNotifierWrappers in |serialized_packet| are swapped from the
35  // serialized packet into the QuicTransmissionInfo.
36  void AddSentPacket(SerializedPacket* serialized_packet,
37  QuicPacketNumber old_packet_number,
38  TransmissionType transmission_type,
39  QuicTime sent_time,
40  bool set_in_flight);
41 
42  // Returns true if the packet |packet_number| is unacked.
43  bool IsUnacked(QuicPacketNumber packet_number) const;
44 
45  // Notifies session_notifier that frames have been acked. Returns true if any
46  // new data gets acked, returns false otherwise.
47  bool NotifyFramesAcked(const QuicTransmissionInfo& info,
48  QuicTime::Delta ack_delay);
49 
50  // Notifies session_notifier that frames in |info| are considered as lost.
51  void NotifyFramesLost(const QuicTransmissionInfo& info,
52  TransmissionType type);

59  // Marks |info| as no longer in flight.

62  // Marks |packet_number| as no longer in flight.

64 
65  // No longer retransmit data for |stream_id|.
66  void CancelRetransmissionsForStream(QuicStreamId stream_id);

68  // Returns true if |packet_number| has retransmittable frames. This will
69  // return false if all frames of this packet are either non-retransmittable or

200  // Newly serialized retransmittable packets are added to this map, which
201  // contains owning pointers to any contained frames. If a packet is
202  // retransmitted, this map will contain entries for both the old and the new
203  // packet. The old packet's retransmittable frames entry will be nullptr,
204  // while the new packet's entry will contain the frames to retransmit.
205  // If the old packet is acked before the new packet, then the old entry will
206  // be removed from the map and the new entry's retransmittable frames will be
207  // set to nullptr.
208  UnackedPacketMap unacked_packets_;
209  // The packet at the 0th index of unacked_packets_.
210  QuicPacketNumber least_unacked_;

215

218 
219  // If true, let session decides what to write.
220  bool session_decides_what_to_write_;
