45 class QUIC_EXPORT_PRIVATE QuicCryptoClientStream
46  : public QuicCryptoClientStreamBase {
47  public:
48  // kMaxClientHellos is the maximum number of times that we'll send a client
49  // hello. The value 3 accounts for:
50  // * One failure due to an incorrect or missing source-address token.
51  // * One failure due the server's certificate chain being unavailible and
52  // the server being unwilling to send it without a valid source-address
53  // token.
54  static const int kMaxClientHellos = 3;
55 
56  // QuicCryptoClientStream creates a HandshakerDelegate at construction time
57  // based on the QuicTransportVersion of the connection. Different
58  // HandshakerDelegates provide implementations of different crypto handshake
59  // protocols. Currently QUIC crypto is the only protocol implemented; a future
60  // HandshakerDelegate will use TLS as the handshake protocol.
61  // QuicCryptoClientStream delegates all of its public methods to its
62  // HandshakerDelegate.
63  //
64  // This setup of the crypto stream delegating its implementation to the
65  // handshaker results in the handshaker reading and writing bytes on the
66  // crypto stream, instead of the handshaker passing the stream bytes to send.
67  class QUIC_EXPORT_PRIVATE HandshakerDelegate {
68  public:
69  virtual ~HandshakerDelegate() {}
70 
71  // Performs a crypto handshake with the server. Returns true if the
72  // connection is still connected.
73  virtual bool CryptoConnect() = 0;
74 
75  // num_sent_client_hellos returns the number of client hello messages that
76  // have been sent. If the handshake has completed then this is one greater
77  // than the number of round-trips needed for the handshake.
78  virtual int num_sent_client_hellos() const = 0;
79 
80  // The number of server config update messages received by the
81  // client. Does not count update messages that were received prior
82  // to handshake confirmation.
83  virtual int num_scup_messages_received() const = 0;
84 
85  // Returns true if a channel ID was sent on this connection.
86  virtual bool WasChannelIDSent() const = 0;
87 
88  // Returns true if our ChannelIDSourceCallback was run, which implies the
89  // ChannelIDSource operated asynchronously. Intended for testing.
90  virtual bool WasChannelIDSourceCallbackRun() const = 0;
91 
92  virtual QuicString chlo_hash() const = 0;
93 
94  // Returns true once any encrypter (initial/0RTT or final/1RTT) has been set
95  // for the connection.
96  virtual bool encryption_established() const = 0;
97 
98  // Returns true once the crypto handshake has completed.
99  virtual bool handshake_confirmed() const = 0;
100 
101  // Returns the parameters negotiated in the crypto handshake.
102  virtual const QuicCryptoNegotiatedParameters& crypto_negotiated_params()
103  const = 0;
104 
105  // Used by QuicCryptoStream to parse data received on this stream.
106  virtual CryptoMessageParser* crypto_message_parser() = 0;
107  };
