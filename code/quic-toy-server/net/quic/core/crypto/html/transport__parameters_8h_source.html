15 // TransportParameters contains parameters for QUIC's transport layer that are
16 // indicated during the TLS handshake. This struct is a mirror of the struct in
17 // section 7.4 of draft-ietf-quic-transport-08.
18 struct QUIC_EXPORT_PRIVATE TransportParameters {
19  TransportParameters();
20  TransportParameters(const TransportParameters& transport_params);
21  ~TransportParameters();
22 
23  // When |perspective| is Perspective::IS_CLIENT, this struct is being used in
24  // the client_hello handshake message; when it is Perspective::IS_SERVER, it
25  // is being used in the encrypted_extensions handshake message.
26  Perspective perspective;
27 
28  // When Perspective::IS_CLIENT, |version| is the initial version offered by
29  // the client (before any version negotiation packets) for this connection.
30  // When Perspective::IS_SERVER, |version| is the version that is in use.
31  QuicVersionLabel version;
32 
33  // Server-only parameters:
34 
35  // |supported_versions| contains a list of all versions that the server would
36  // send in a version negotiation packet. It is not used if |perspective ==
37  // Perspective::IS_CLIENT|.
38  QuicVersionLabelVector supported_versions;
39 
40  // See section 7.4.1 of draft-ietf-quic-transport-08 for definition.
41  std::vector<uint8_t> stateless_reset_token;
42 
43  // Required parameters. See section 7.4.1 of draft-ietf-quic-transport-08 for
44  // definitions.
45  uint32_t initial_max_stream_data = 0;
46  uint32_t initial_max_data = 0;
47  uint16_t idle_timeout = 0;
48 
49  template <typename T>
50  struct OptionalParam {
51  bool present = false;
52  T value;
53  };
54 
55  // Optional parameters. See section 7.4.1 of draft-ietf-quic-transport-08 for
56  // definitions.
57  OptionalParam<uint32_t> initial_max_stream_id_bidi;
58  OptionalParam<uint32_t> initial_max_stream_id_uni;
59  OptionalParam<uint16_t> max_packet_size;
60  OptionalParam<uint8_t> ack_delay_exponent;
61  bool omit_connection_id = false;
62 
63  // Returns true if the contents of this struct are valid.
64  bool is_valid() const;
65 };

67 // Serializes a TransportParameters struct into the format for sending it in a
68 // TLS extension. The serialized bytes are put in |*out|, and this function
69 // returns true on success or false if |TransportParameters::is_valid| returns
70 // false.
71 QUIC_EXPORT_PRIVATE bool SerializeTransportParameters(
72  const TransportParameters& in,
73  std::vector<uint8_t>* out);

75 // Parses bytes from the quic_transport_parameters TLS extension and writes the
76 // parsed parameters into |*out|. Input is read from |in| for |in_len| bytes.
77 // |perspective| indicates whether the input came from a client or a server.
78 // This method returns true if the input was successfully parsed, and false if
79 // it could not be parsed.
80 // TODO(nharper): Write fuzz tests for this method.
81 QUIC_EXPORT_PRIVATE bool ParseTransportParameters(const uint8_t* in,
82  size_t in_len,
83  Perspective perspective,
84  TransportParameters* out);
