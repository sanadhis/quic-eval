38 // A class for framing the crypto messages that are exchanged in a QUIC
39 // session.
40 class QUIC_EXPORT_PRIVATE CryptoFramer : public CryptoMessageParser {
41  public:
42  CryptoFramer();
43 
44  ~CryptoFramer() override;
45 
46  // ParseMessage parses exactly one message from the given QuicStringPiece. If
47  // there is an error, the message is truncated, or the message has trailing
48  // garbage then nullptr will be returned.
49  static std::unique_ptr<CryptoHandshakeMessage> ParseMessage(
50  QuicStringPiece in,
51  Perspective perspective);
52 
53  // Set callbacks to be called from the framer. A visitor must be set, or
54  // else the framer will crash. It is acceptable for the visitor to do
55  // nothing. If this is called multiple times, only the last visitor
56  // will be used. |visitor| will be owned by the framer.
57  void set_visitor(CryptoFramerVisitorInterface* visitor) {
58  visitor_ = visitor;
59  }
60 
61  QuicErrorCode error() const override;
62  const QuicString& error_detail() const override;
63 
64  // Processes input data, which must be delivered in order. Returns
65  // false if there was an error, and true otherwise.
66  bool ProcessInput(QuicStringPiece input, Perspective perspective) override;
67 
68  // Returns the number of bytes of buffered input data remaining to be
69  // parsed.
70  size_t InputBytesRemaining() const override;
71 
72  // Checks if the specified tag has been seen. Returns |true| if it
73  // has, and |false| if it has not or a CHLO has not been seen.
74  bool HasTag(QuicTag tag) const;
75 
76  // Even if the CHLO has not been fully received, force processing of
77  // the handshake message. This is dangerous and should not be used
78  // except as a mechanism of last resort.
79  void ForceHandshake();
80 
81  // Returns a new QuicData owned by the caller that contains a serialized
82  // |message|, or nullptr if there was an error.
83  static QuicData* ConstructHandshakeMessage(
84  const CryptoHandshakeMessage& message,
85  Perspective perspective);
86 
87  private:
88  // Clears per-message state. Does not clear the visitor.
89  void Clear();
90 
91  // Process does does the work of |ProcessInput|, but returns an error code,
92  // doesn't set error_ and doesn't call |visitor_->OnError()|.
93  QuicErrorCode Process(QuicStringPiece input, Perspective perspective);
94 
95  static bool WritePadTag(QuicDataWriter* writer,
96  size_t pad_length,
97  uint32_t* end_offset);
98 
99  // Represents the current state of the parsing state machine.
100  enum CryptoFramerState {
101  STATE_READING_TAG,
102  STATE_READING_NUM_ENTRIES,
103  STATE_READING_TAGS_AND_LENGTHS,
104  STATE_READING_VALUES
105  };
106 
107  // Visitor to invoke when messages are parsed.
108  CryptoFramerVisitorInterface* visitor_;
109  // Last error.
110  QuicErrorCode error_;
111  // Remaining unparsed data.
112  QuicString buffer_;
113  // Current state of the parsing.
114  CryptoFramerState state_;
115  // The message currently being parsed.
116  CryptoHandshakeMessage message_;
117  // The issue which caused |error_|
118  QuicString error_detail_;
119  // Number of entires in the message currently being parsed.
120  uint16_t num_entries_;
121  // tags_and_lengths_ contains the tags that are currently being parsed and
122  // their lengths.
123  std::vector<std::pair<QuicTag, size_t>> tags_and_lengths_;
124  // Cumulative length of all values in the message currently being parsed.
125  size_t values_len_;
126 };
