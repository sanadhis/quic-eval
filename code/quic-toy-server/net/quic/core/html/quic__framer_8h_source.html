60 // This class receives callbacks from the framer when packets
61 // are processed.
62 class QUIC_EXPORT_PRIVATE QuicFramerVisitorInterface {
63  public:
64  virtual ~QuicFramerVisitorInterface() {}
65 
66  // Called if an error is detected in the QUIC protocol.
67  virtual void OnError(QuicFramer* framer) = 0;
68 
69  // Called only when |perspective_| is IS_SERVER and the framer gets a
70  // packet with version flag true and the version on the packet doesn't match
71  // |quic_version_|. The visitor should return true after it updates the
72  // version of the |framer_| to |received_version| or false to stop processing
73  // this packet.
74  virtual bool OnProtocolVersionMismatch(
75  ParsedQuicVersion received_version) = 0;
76 
77  // Called when a new packet has been received, before it
78  // has been validated or processed.
79  virtual void OnPacket() = 0;
80 
81  // Called when a public reset packet has been parsed but has not yet
82  // been validated.
83  virtual void OnPublicResetPacket(const QuicPublicResetPacket& packet) = 0;
84 
85  // Called only when |perspective_| is IS_CLIENT and a version negotiation
86  // packet has been parsed.
87  virtual void OnVersionNegotiationPacket(
88  const QuicVersionNegotiationPacket& packet) = 0;
89 
90  // Called when all fields except packet number has been parsed, but has not
91  // been authenticated. If it returns false, framing for this packet will
92  // cease.
93  virtual bool OnUnauthenticatedPublicHeader(
94  const QuicPacketHeader& header) = 0;
95 
96  // Called when the unauthenticated portion of the header has been parsed.
97  // If OnUnauthenticatedHeader returns false, framing for this packet will
98  // cease.
99  virtual bool OnUnauthenticatedHeader(const QuicPacketHeader& header) = 0;
100 
101  // Called when a packet has been decrypted. |level| is the encryption level
102  // of the packet.
103  virtual void OnDecryptedPacket(EncryptionLevel level) = 0;
104 
105  // Called when the complete header of a packet had been parsed.
106  // If OnPacketHeader returns false, framing for this packet will cease.
107  virtual bool OnPacketHeader(const QuicPacketHeader& header) = 0;
108 
109  // Called when a StreamFrame has been parsed.
110  virtual bool OnStreamFrame(const QuicStreamFrame& frame) = 0;
111 
112  // Called when a AckFrame has been parsed. If OnAckFrame returns false,
113  // the framer will stop parsing the current packet.
114  virtual bool OnAckFrame(const QuicAckFrame& frame) = 0;
115 
116  // Called when a StopWaitingFrame has been parsed.
117  virtual bool OnStopWaitingFrame(const QuicStopWaitingFrame& frame) = 0;
118 
119  // Called when a QuicPaddingFrame has been parsed.
120  virtual bool OnPaddingFrame(const QuicPaddingFrame& frame) = 0;
121 
122  // Called when a PingFrame has been parsed.
123  virtual bool OnPingFrame(const QuicPingFrame& frame) = 0;
124 
125  // Called when a RstStreamFrame has been parsed.
126  virtual bool OnRstStreamFrame(const QuicRstStreamFrame& frame) = 0;
127 
128  // Called when a ConnectionCloseFrame has been parsed.
129  virtual bool OnConnectionCloseFrame(
130  const QuicConnectionCloseFrame& frame) = 0;
131 
132  // Called when a GoAwayFrame has been parsed.
133  virtual bool OnGoAwayFrame(const QuicGoAwayFrame& frame) = 0;
134 
135  // Called when a WindowUpdateFrame has been parsed.
136  virtual bool OnWindowUpdateFrame(const QuicWindowUpdateFrame& frame) = 0;
137 
138  // Called when a BlockedFrame has been parsed.
139  virtual bool OnBlockedFrame(const QuicBlockedFrame& frame) = 0;
140 
141  // Called when a packet has been completely processed.
142  virtual void OnPacketComplete() = 0;
143 };
