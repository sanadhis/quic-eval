18 // QuicTlsAdapter provides an implementation of CryptoMessageParser that takes
19 // incoming messages and provides them to be read in a BIO (used by the TLS
20 // stack to read incoming messages). Messages written to the BIO by the TLS
21 // stack are provided to the QuicTlsAdapter's consumer through the
22 // OnDataReceived method of the consumer's implementation of
23 // QuicTlsAdapter::Visitor.
24 //
25 // QuicTlsAdapter also provides an implementation of the BIO interface,
26 // openssl's abstraction used by the TLS stack for I/O. The BIO interface
27 // provides BIO_read, BIO_write, and BIO_flush methods, with an API very similar
28 // to Berkeley sockets. This is a non-blocking interface - if data is not
29 // available for the BIO consumer to read with BIO_read, it returns 0 bytes of
30 // data, and the BIO consumer must handle waiting for more data and only calling
31 // BIO_read once data is available to read. With a QuicTlsAdapter, the signal
32 // that data is available to read is provided by
33 // QuicTlsAdapter::Visitor::OnDataAvailableForBIO.
34 //
35 // In effect, the QuicTlsAdapter moves messages between the QuicCryptoStream and
36 // the TLS stack. On one end, the QuicTlsAdapter implements CryptoMessageParser
37 // to take incoming messages and make them available to be read through the BIO,
38 // and on the other end, it takes messages written to the BIO and once the BIO
39 // flushes them, sends them out to the QuicStream via the
40 // QuicTlsAdapter::Visitor.
41 //
42 // Data flows from a QuicCryptoStream to the TLS stack like so:
43 // 1. QuicCryptoStream::OnDataAvailable is called (by QuicStream) when data is
44 // available on the stream.
45 // 2. OnDataAvailable calls CryptoMessageParser::ProcessInput; in the case of a
46 // TLS crypto stream, this is QuicTlsAdapter::ProcessInput.
47 // 3. ProcessInput saves the data to QuicTlsAdapter's read buffer, and signals
48 // that data is available to read by calling
49 // QuicTlsAdapter::Visitor::OnDataAvailableForBIO.
50 // 4. TlsHandshaker (which implements QuicTlsAdapter::Visitor) receives the
51 // OnDataAvailableForBIO, and has the TLS stack continue its handshake.
52 // 5. The TLS stack calls BIO_read to read handshake messages, and this call is
53 // made on a BIO backed by QuicTlsAdapter.
54 // 6. BIO_read calls QuicTlsAdapter::BIOReadWrapper, which calls
55 // QuicTlsAdapter::Read (on the appropriate instance) which provides the
56 // data from the read buffer written to by QuicTlsAdapter::ProcessInput.
57 //
58 // Data flows from the TLS stack to the QUIC crypto stream like so:
59 // 1. The TLS stack makes multiple calls to BIO_write as it generates handshake
60 // messages. Via QuicTlsAdapter::BIOWriteWrapper and QuicTlsAdapter::Write,
61 // this data gets appended to the QuicTlsAdapter's write buffer.
62 // 2. Once the TLS stack has written a flight of handshake messages, it calls
63 // BIO_flush. This, via QuicTlsAdapter::BIOCtrlWrapper and
64 // QuicTlsAdapter::Flush, signals to QuicTlsAdapter's Visitor that data has
65 // been received.
66 // 3. QuicTlsAdapter::Flush calls
67 // QuicTlsAdapter::Visitor::OnDataReceivedFromBIO with the contents of the
68 // write buffer.
69 // 4. TlsHandshaker receives the data from OnDataReceivedFromBIO, and writes it
70 // to the QUIC crypto stream.
71 class QUIC_EXPORT QuicTlsAdapter : public CryptoMessageParser {
72  public:
73  // QuicTlsAdapter::Visitor is notified whenever data is received (in either
74  // direction). When data is read from the QUIC crypto stream,
75  // Visitor::OnDataAvailableForBIO is called so that the Visitor can continue
76  // reading from the BIO. (E.g. in the case of a TlsHandshaker as the Visitor,
77  // it would continue the TLS handshake, which uses a BIO for I/O.) When data
78  // is written to a QuicTlsAdapter's BIO interface and then flushed,
79  // Visitor::OnDataReceivedFromBIO is called to provide the Visitor with the
80  // data to write to the QUIC crypto stream.
81  class QUIC_EXPORT Visitor {
82  public:
83  virtual ~Visitor() {}
84 
85  // OnDataAvailableForBIO is called when QuicTlsAdapter has received data
86  // (via ProcessInput) that is now available to be read by the BIO.
87  virtual void OnDataAvailableForBIO() = 0;
88 
89  // OnDataReceivedFromBIO is called when data is written to the BIO. For
90  // example, when the TLS stack writes messages to the BIO and then flushes
91  // them, the resulting data will be made available to the Visitor via this
92  // method, so that the Visitor can write the messages to the QuicStream. The
93  // stringpiece |data| is only valid during the execution of this function;
94  // implementations must consume all of |data|.
95  virtual void OnDataReceivedFromBIO(const QuicStringPiece& data) = 0;
96  };
97 
98  // Constructs a QuicTlsAdapter that will notify |visitor| when data is
99  // available in either direction. The provided Visitor must outlive the
100  // QuicTlsAdapter.
101  explicit QuicTlsAdapter(Visitor* visitor);
102 
103  ~QuicTlsAdapter() override;

107  bool ProcessInput(QuicStringPiece input, Perspective perspective) override;
108  size_t InputBytesRemaining() const override;

110  BIO* bio() { return bio_.get(); }
111 
112  private:
113  // The following methods, Read, Write, and Flush, are used to implement the
114  // BIO.
115 
116  // Read copies up to |len| bytes from |read_buffer_| into |out|. It returns
117  // the number of bytes copied, zero on EOF, or a negative number on error.
118  int Read(char* out, int len);
119 
120  // Write appends |len| bytes from |in| to |write_buffer_|. It returns |len|,
121  // or a negative number on error.
122  int Write(const char* in, int len);
123 
124  // Flush calls Visitor::OnDataReceivedFromBIO with the data in
125  // |write_buffer_| and then empties the buffer.
126  void Flush();

128  // Used by the static BIO*Wrapper methods to get the QuicTlsAdapter instance
129  // to call Read/Write/Flush on.
130  static QuicTlsAdapter* GetAdapter(BIO* bio);
131 
132  // Functions used to build a BIO_METHOD vtable. BIOReadWrapper calls Read,
133  // BIOWriteWrapper calls Write, and BIOCtrlWrapper calls Flush if |cmd| is
134  // |BIO_CTRL_FLUSH|.
135 
136  static int BIOReadWrapper(BIO* bio, char* out, int len);
137  static int BIOWriteWrapper(BIO* bio, const char* in, int len);
138  // BIOCtrlWrapper has the type it does so it can be used as the |ctrl| field
139  // in a BIO_METHOD struct, hence the use of long as an argument and return
140  // type.
141  // NOLINTNEXTLINE
142  static long BIOCtrlWrapper(BIO* bio, int cmd, long larg, void* parg);
143

155  QuicString write_buffer_;
