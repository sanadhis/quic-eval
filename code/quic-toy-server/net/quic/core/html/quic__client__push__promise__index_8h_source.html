17 // QuicClientPushPromiseIndex is the interface to support rendezvous
18 // between client requests and resources delivered via server push.
19 // The same index can be shared across multiple sessions (e.g. for the
20 // same browser users profile), since cross-origin pushes are allowed
21 // (subject to authority constraints).
22 
23 class QUIC_EXPORT_PRIVATE QuicClientPushPromiseIndex {
24  public:
25  // Delegate is used to complete the rendezvous that began with
26  // |Try()|.
27  class QUIC_EXPORT_PRIVATE Delegate {
28  public:
29  virtual ~Delegate() {}
30 
31  // The primary lookup matched request with push promise by URL. A
32  // secondary match is necessary to ensure Vary (RFC 2616, 14.14)
33  // is honored. If Vary is not present, return true. If Vary is
34  // present, return whether designated header fields of
35  // |promise_request| and |client_request| match.
36  virtual bool CheckVary(const SpdyHeaderBlock& client_request,
37  const SpdyHeaderBlock& promise_request,
38  const SpdyHeaderBlock& promise_response) = 0;
39 
40  // On rendezvous success, provides the promised |stream|. Callee
41  // does not inherit ownership of |stream|. On rendezvous failure,
42  // |stream| is |nullptr| and the client should retry the request.
43  // Rendezvous can fail due to promise validation failure or RST on
44  // promised stream. |url| will have been removed from the index
45  // before |OnRendezvousResult()| is invoked, so a recursive call to
46  // |Try()| will return |QUIC_FAILURE|, which may be convenient for
47  // retry purposes.
48  virtual void OnRendezvousResult(QuicSpdyStream* stream) = 0;
49  };

53  // Cancel the request.
54  virtual void Cancel() = 0;
55 
56  protected:
57  TryHandle() {}
58  ~TryHandle();
59 
60  private:
61  DISALLOW_COPY_AND_ASSIGN(TryHandle);
62  };

64  QuicClientPushPromiseIndex();
65  virtual ~QuicClientPushPromiseIndex();
66 
67  // Called by client code, used to enforce affinity between requests
68  // for promised streams and the session the promise came from.
69  QuicClientPromisedInfo* GetPromised(const QuicString& url);
70 
71  // Called by client code, to initiate rendezvous between a request
72  // and a server push stream. If |request|'s url is in the index,
73  // rendezvous will be attempted and may complete immediately or
74  // asynchronously. If the matching promise and response headers
75  // have already arrived, the delegate's methods will fire
76  // recursively from within |Try()|. Returns |QUIC_SUCCESS| if the
77  // rendezvous was a success. Returns |QUIC_FAILURE| if there was no
78  // matching promise, or if there was but the rendezvous has failed.
79  // Returns QUIC_PENDING if a matching promise was found, but the
80  // rendezvous needs to complete asynchronously because the promised
81  // response headers are not yet available. If result is
82  // QUIC_PENDING, then |*handle| will set so that the caller may
83  // cancel the request if need be. The caller does not inherit
84  // ownership of |*handle|, and it ceases to be valid if the caller
85  // invokes |handle->Cancel()| or if |delegate->OnReponse()| fires.
86  QuicAsyncStatus Try(const SpdyHeaderBlock& request,
87  Delegate* delegate,
88  TryHandle** handle);
89 
90  QuicPromisedByUrlMap* promised_by_url() { return &promised_by_url_; }
91 
92  private:
93  QuicPromisedByUrlMap promised_by_url_;
94 
95  DISALLOW_COPY_AND_ASSIGN(QuicClientPushPromiseIndex);
96 };
