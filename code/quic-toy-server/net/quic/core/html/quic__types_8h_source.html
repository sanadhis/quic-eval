161  // the low-3 bits of the stream frame type value are actually flags

208  // Bit 0: Does the packet header contains version info?

214  // Bit 2: indicates the header includes a nonce.

216 
217  // Bit 3: indicates whether a ConnectionID is included.
218  PACKET_PUBLIC_FLAGS_0BYTE_CONNECTION_ID = 0,
219  PACKET_PUBLIC_FLAGS_8BYTE_CONNECTION_ID = 1 << 3,
220 
221  // QUIC_VERSION_32 and earlier use two bits for an 8 byte
222  // connection id.
223  PACKET_PUBLIC_FLAGS_8BYTE_CONNECTION_ID_OLD = 1 << 3 | 1 << 2,
224 
225  // Bits 4 and 5 describe the packet number length as follows:
226  // --00----: 1 byte
227  // --01----: 2 bytes
228  // --10----: 4 bytes
229  // --11----: 6 bytes
230  PACKET_PUBLIC_FLAGS_1BYTE_PACKET = PACKET_FLAGS_1BYTE_PACKET << 4,
231  PACKET_PUBLIC_FLAGS_2BYTE_PACKET = PACKET_FLAGS_2BYTE_PACKET << 4,
232  PACKET_PUBLIC_FLAGS_4BYTE_PACKET = PACKET_FLAGS_4BYTE_PACKET << 4,
233  PACKET_PUBLIC_FLAGS_6BYTE_PACKET = PACKET_FLAGS_8BYTE_PACKET << 4,

236 
237  // Bit 7: indicates the presence of a second flags byte.
