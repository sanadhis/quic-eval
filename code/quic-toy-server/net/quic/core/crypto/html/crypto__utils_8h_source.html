30  // Diversification is a utility class that's used to act like a union type.
31  // Values can be created by calling the functions like |NoDiversification|,
32  // below.
33  class Diversification {
34  public:
35  enum Mode {
36  NEVER, // Key diversification will never be used. Forward secure
37  // crypters will always use this mode.
38 
39  PENDING, // Key diversification will happen when a nonce is later
40  // received. This should only be used by clients initial
41  // decrypters which are waiting on the divesification nonce
42  // from the server.
43 
44  NOW, // Key diversification will happen immediate based on the nonce.
45  // This should only be used by servers initial encrypters.
46  };
47 
48  Diversification(const Diversification& diversification) = default;
49 
50  static Diversification Never() { return Diversification(NEVER, nullptr); }
51  static Diversification Pending() {
52  return Diversification(PENDING, nullptr);
53  }
54  static Diversification Now(DiversificationNonce* nonce) {
55  return Diversification(NOW, nonce);
56  }
57 
58  Mode mode() const { return mode_; }
59  DiversificationNonce* nonce() const {
60  DCHECK_EQ(mode_, NOW);
61  return nonce_;
62  }
63 
64  private:
65  Diversification(Mode mode, DiversificationNonce* nonce)
66  : mode_(mode), nonce_(nonce) {}
67 
68  Mode mode_;
69  DiversificationNonce* nonce_;
70  };

97  // DeriveKeys populates |crypters->encrypter|, |crypters->decrypter|, and
98  // |subkey_secret| (optional -- may be null) given the contents of
99  // |premaster_secret|, |client_nonce|, |server_nonce| and |hkdf_input|. |aead|
100  // determines which cipher will be used. |perspective| controls whether the
101  // server's keys are assigned to |encrypter| or |decrypter|. |server_nonce| is
102  // optional and, if non-empty, is mixed into the key derivation.
103  // |subkey_secret| will have the same length as |premaster_secret|.
104  //
105  // If the mode of |diversification| is NEVER, the the crypters will be
106  // configured to never perform key diversification. If the mode is
107  // NOW (which is only for servers, then the encrypter will be keyed via a
108  // two-step process that uses the nonce from |diversification|.
109  // If the mode is PENDING (which is only for servres), then the
110  // decrypter will only be keyed to a preliminary state: a call to
111  // |SetDiversificationNonce| with a diversification nonce will be needed to
112  // complete keying.
113  static bool DeriveKeys(QuicStringPiece premaster_secret,
114  QuicTag aead,
115  QuicStringPiece client_nonce,
116  QuicStringPiece server_nonce,
117  const QuicString& hkdf_input,
118  Perspective perspective,
119  Diversification diversification,
120  CrypterPair* crypters,
121  QuicString* subkey_secret);
