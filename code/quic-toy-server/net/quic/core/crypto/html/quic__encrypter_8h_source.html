23  // Creates an IETF QuicEncrypter based on |cipher_suite| which must be an id
24  // returned by SSL_CIPHER_get_id. The caller is responsible for taking

28  // Sets the encryption key. Returns true on success, false on failure.
29  //
30  // NOTE: The key is the client_write_key or server_write_key derived from
31  // the master secret.
32  virtual bool SetKey(QuicStringPiece key) = 0;
33 
34  // Sets the fixed initial bytes of the nonce. Returns true on success,
35  // false on failure. This method must only be used with Google QUIC crypters.
36  //
37  // NOTE: The nonce prefix is the client_write_iv or server_write_iv
38  // derived from the master secret. A 64-bit packet number will
39  // be appended to form the nonce.
40  //
41  // <------------ 64 bits ----------->
42  // +---------------------+----------------------------------+
43  // | Fixed prefix | packet number |
44  // +---------------------+----------------------------------+
45  // Nonce format
46  //
47  // The security of the nonce format requires that QUIC never reuse a
48  // packet number, even when retransmitting a lost packet.
49  virtual bool SetNoncePrefix(QuicStringPiece nonce_prefix) = 0;
50 
51  // Sets |iv| as the initialization vector to use when constructing the nonce.
52  // Returns true on success, false on failure. This method must only be used
53  // with IETF QUIC crypters.
54  //
55  // Google QUIC and IETF QUIC use different nonce constructions. This method
56  // must be used when using IETF QUIC; SetNoncePrefix must be used when using
57  // Google QUIC.
58  //
59  // The nonce is constructed as follows (draft-ietf-quic-tls section 5.3):
60  //
61  // <---------------- max(8, N_MIN) bytes ----------------->
62  // +--------------------------------------------------------+
63  // | packet protection IV |
64  // +--------------------------------------------------------+
65  // XOR
66  // <------------ 64 bits ----------->
67  // +---------------------+----------------------------------+
68  // | zeroes | reconstructed packet number |
69  // +---------------------+----------------------------------+
70  //
71  // The nonce is the packet protection IV (|iv|) XOR'd with the left-padded
72  // reconstructed packet number.
73  //
74  // The security of the nonce format requires that QUIC never reuse a
75  // packet number, even when retransmitting a lost packet.
76  virtual bool SetIV(QuicStringPiece iv) = 0;
77 
78  // Writes encrypted |plaintext| and a MAC over |plaintext| and
79  // |associated_data| into output. Sets |output_length| to the number of
80  // bytes written. Returns true on success or false if there was an error.
81  // |packet_number| is appended to the |nonce_prefix| value provided in
82  // SetNoncePrefix() to form the nonce. |output| must not overlap with
83  // |associated_data|. If |output| overlaps with |plaintext| then
84  // |plaintext| must be <= |output|.
85  virtual bool EncryptPacket(QuicTransportVersion version,
86  QuicPacketNumber packet_number,
87  QuicStringPiece associated_data,
88  QuicStringPiece plaintext,
89  char* output,
90  size_t* output_length,
91  size_t max_output_length) = 0;
92 
93  // GetKeySize() and GetNoncePrefixSize() tell the HKDF class how many bytes
94  // of key material needs to be derived from the master secret.
95  // NOTE: the sizes returned by GetKeySize() and GetNoncePrefixSize() are
96  // also correct for the QuicDecrypter of the same algorithm. So only
97  // QuicEncrypter has these two methods.
98 
99  // Returns the size in bytes of a key for the algorithm.
100  virtual size_t GetKeySize() const = 0;
101  // Returns the size in bytes of the fixed initial part of the nonce.
102  virtual size_t GetNoncePrefixSize() const = 0;
103 
104  // Returns the size in bytes of an IV to use with the algorithm.
105  virtual size_t GetIVSize() const = 0;
106 
107  // Returns the maximum length of plaintext that can be encrypted
108  // to ciphertext no larger than |ciphertext_size|.
109  virtual size_t GetMaxPlaintextSize(size_t ciphertext_size) const = 0;
110 
111  // Returns the length of the ciphertext that would be generated by encrypting
112  // to plaintext of size |plaintext_size|.
113  virtual size_t GetCiphertextSize(size_t plaintext_size) const = 0;
114 
115  // For use by unit tests only.
116  virtual QuicStringPiece GetKey() const = 0;
117  virtual QuicStringPiece GetNoncePrefix() const = 0;
118 };
