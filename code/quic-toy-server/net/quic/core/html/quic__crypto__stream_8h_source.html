24 // Crypto handshake messages in QUIC take place over a reserved stream with the
25 // id 1. Each endpoint (client and server) will allocate an instance of a
26 // subclass of QuicCryptoStream to send and receive handshake messages. (In the
27 // normal 1-RTT handshake, the client will send a client hello, CHLO, message.
28 // The server will receive this message and respond with a server hello message,
29 // SHLO. At this point both sides will have established a crypto context they
30 // can use to send encrypted messages.
31 //
32 // For more details:
33 // https://docs.google.com/document/d/1g5nIXAIkN_Y-7XJW5K45IblHd_L2f5LTaDUDwvZ5L6g/edit?usp=sharing
34 class QUIC_EXPORT_PRIVATE QuicCryptoStream : public QuicStream {

38  ~QuicCryptoStream() override;

40  // Returns the per-packet framing overhead associated with sending a

43  QuicTransportVersion version);

45  // QuicStream implementation
46  void OnDataAvailable() override;

48  // Performs key extraction to derive a new secret of |result_len| bytes
49  // dependent on |label|, |context|, and the stream's negotiated subkey secret.
50  // Returns false if the handshake has not been confirmed or the parameters are
51  // invalid (e.g. |label| contains null bytes); returns true on success.
52  bool ExportKeyingMaterial(QuicStringPiece label,
53  QuicStringPiece context,
54  size_t result_len,
55  QuicString* result) const;
56 
57  // Performs key extraction for Token Binding. Unlike ExportKeyingMaterial,
58  // this function can be called before forward-secure encryption is
59  // established. Returns false if initial encryption has not been established,
60  // and true on success.
61  //
62  // Since this depends only on the initial keys, a signature over it can be
63  // repurposed by an attacker who obtains the client's or server's DH private
64  // value.
65  bool ExportTokenBindingKeyingMaterial(QuicString* result) const;

97  // Override to send unacked crypto data with the appropriate encryption level.
98  bool RetransmitStreamData(QuicStreamOffset offset,
99  QuicByteCount data_length,
100  bool fin) override;
101 
102  private:

107
