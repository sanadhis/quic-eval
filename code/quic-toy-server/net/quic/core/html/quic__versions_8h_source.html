17 // The available versions of QUIC. Guaranteed that the integer value of the enum
18 // will match the version number.
19 // When adding a new version to this enum you should add it to
20 // kSupportedTransportVersions (if appropriate), and also add a new case to the
21 // helper methods QuicVersionToQuicVersionLabel, QuicVersionLabelToQuicVersion,
22 // and QuicVersionToString.
23 enum QuicTransportVersion {
24  // Special case to indicate unknown/unsupported QUIC version.
25  QUIC_VERSION_UNSUPPORTED = 0,
26 
27  QUIC_VERSION_35 = 35, // Allows endpoints to independently set stream limit.
28  QUIC_VERSION_37 = 37, // Add perspective into null encryption.
29  QUIC_VERSION_38 = 38, // PADDING frame is a 1-byte frame with type 0x00.
30  // Respect NSTP connection option.
31  QUIC_VERSION_39 = 39, // Integers and floating numbers are written in big
32  // endian. Dot not ack acks. Send a connection level
33  // WINDOW_UPDATE every 20 sent packets which do not
34  // contain retransmittable frames.
35  QUIC_VERSION_41 = 41, // RST_STREAM, ACK and STREAM frames match IETF format.
36  QUIC_VERSION_42 = 42, // Allows receiving overlapping stream data.
37  QUIC_VERSION_43 = 43, // PRIORITY frames are sent by client and accepted by
38  // server.
39  QUIC_VERSION_99 = 99, // Dumping ground for IETF QUIC changes which are not
40  // yet ready for production.
41 
42  // IMPORTANT: if you are adding to this list, follow the instructions at
43  // http://sites/quic/adding-and-removing-versions
44 };

46 // The crypto handshake protocols that can be used with QUIC.
47 enum HandshakeProtocol {
48  PROTOCOL_UNSUPPORTED,
49  PROTOCOL_QUIC_CRYPTO,
50  PROTOCOL_TLS1_3,
51 };

53 // A parsed QUIC version label which determines that handshake protocol
54 // and the transport version.
55 struct ParsedQuicVersion {
56  HandshakeProtocol handshake_protocol;
57  QuicTransportVersion transport_version;
58 
59  ParsedQuicVersion(HandshakeProtocol handshake_protocol,
60  QuicTransportVersion transport_version)
61  : handshake_protocol(handshake_protocol),
62  transport_version(transport_version) {}
63 
64  ParsedQuicVersion(const ParsedQuicVersion& other)
65  : handshake_protocol(other.handshake_protocol),
66  transport_version(other.transport_version) {}
67 
68  ParsedQuicVersion& operator=(const ParsedQuicVersion& other) {
69  if (this != &other) {
70  handshake_protocol = other.handshake_protocol;
71  transport_version = other.transport_version;
72  }
73  return *this;
74  }
75 
76  bool operator==(const ParsedQuicVersion& other) const {
77  return handshake_protocol == other.handshake_protocol &&
78  transport_version == other.transport_version;
79  }
80 
81  bool operator!=(const ParsedQuicVersion& other) const {
82  return handshake_protocol != other.handshake_protocol ||
83  transport_version != other.transport_version;
84  }
85 };
